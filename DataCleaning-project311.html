# Function to categorize hours - Less than 2 hours - Excellent, Between 2 to 4 hours - Moderate, 
#Between 4 to 6 - Average, More than 6 hours - Bad
def Ratings(hr):
    if (math.isnan(hr)):
        return 'Unspecified'
    elif (hr < 2.0):
        return 'Excellent'
    elif (4.0 > hr >= 2.0):
        return 'Moderate'
    elif (6.0 > hr >= 4.0):
        return 'Average'
    else:
        return 'Bad'
